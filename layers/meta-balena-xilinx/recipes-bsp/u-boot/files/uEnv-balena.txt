machine_name=zcu102-zynqmp
kernel_load_address=0x80000
devicetree_load_address=0x4000000
kernel_spinor_offset=3c00000
dtb_spinor_offset=5f00000
loadkernel=sf read ${kernel_load_address} ${kernel_spinor_offset} 1e00000
loaddtb=sf read ${devicetree_load_address} ${dtb_spinor_offset} 100000
bootkernel=run loadkernel && run loaddtb && booti ${kernel_load_address} - ${devicetree_load_address}
uenvcmd=sf probe; setenv resin_kernel_load_addr ${kernel_load_address}; run resin_set_kernel_root; setenv bootargs "earlycon clk_ignore_unused ${resin_kernel_root} rw rootwait"; run bootkernel
